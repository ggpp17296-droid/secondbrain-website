```typescript
import { Badge } from '@/components/ui/Badge'
import { Card } from '@/components/ui/Card'
import { ProgressBar } from '@/components/ui/ProgressBar'
import vaultData from '@/../../data/vault-data.json'
import learningStats from '@/../../data/learning-stats.json'

// 任务状态图标
const getStatusIcon = (status: string) => {
  switch (status) {
    case 'completed': return '✅'
    case 'in_progress': return '🔄'
    case 'skipped': return '⏭️'
    default: return '⏳'
  }
}

// 从项目内容中提取任务（简化版，手动定义）
const tasks = [
  { id: 1, title: 'Task 1: 补齐 Context', week: 1, status: 'completed' },
  { id: 2, title: 'Task 2: 提取知识卡片', week: 1, status: 'completed' },
  { id: 3, title: 'Task 3: 规划小项目', week: 1, status: 'completed' },
  { id: 4, title: 'Task 4: Bash 实战', week: 2, status: 'skipped' },
  { id: 5, title: 'Task 5: 设计简单网页', week: 2, status: 'completed' },
  { id: 6, title: 'Task 6: Codex 初探', week: 2, status: 'pending' },
  { id: 7, title: 'Task 7: 信息处理流程', week: 3, status: 'pending' },
  { id: 8, title: 'Task 8: PPT 大纲', week: 4, status: 'pending' },
  { id: 9, title: 'Task 9: 使用手册', week: 4, status: 'pending' },
]

// 按周分组
const tasksByWeek = tasks.reduce((acc, task) => {
  if (!acc[task.week]) acc[task.week] = []
  acc[task.week].push(task)
  return acc
}, {} as Record<number, typeof tasks>)

export default function LearningPage() {
  const stats = learningStats
  const completedTasks = tasks.filter(t => t.status === 'completed').length

  return (
    <div className="mx-auto max-w-7xl px-6 py-12">
      {/* 顶部进度概览 */}
      <div className="mb-12">
        <Badge variant="info" size="md" className="mb-4">
          第 {stats.currentDay} 天 · 第 {stats.currentWeek} 周
        </Badge>
        <h1 className="text-4xl font-bold text-text-primary">
          30 天 Claude Code 学习计划
        </h1>
        <p className="mt-4 text-lg text-text-secondary">
          通过任务驱动 + 案例模仿的方式，系统化掌握 Claude Code 的使用
        </p>
        
        <div className="mt-8 max-w-2xl">
          <ProgressBar 
            value={completedTasks} 
            max={9} 
            label="总体进度"
            size="lg"
          />
        </div>
      </div>

      {/* 时间线展示 */}
      <div className="mb-12">
        <h2 className="mb-6 text-2xl font-semibold text-text-primary">学习时间线</h2>
        <div className="space-y-8">
          {Object.entries(tasksByWeek).map(([week, weekTasks]) => {
            const weekNum = parseInt(week)
            const allCompleted = weekTasks.every(t => t.status === 'completed')
            const hasInProgress = weekTasks.some(t => t.status === 'in_progress')
            
            const weekStatus = allCompleted ? 'completed' : hasInProgress ? 'in_progress' : 'pending'
            const weekLabel = allCompleted ? '已完成' : hasInProgress ? '进行中' : '未开始'
            
            return (
              <div key={week} className="relative">
                {/* 周标题 */}
                <div className="mb-4 flex items-center gap-3">
                  <h3 className="text-xl font-semibold text-text-primary">
                    Week {weekNum}
                  </h3>
                  <Badge 
                    variant={weekStatus === 'completed' ? 'success' : weekStatus === 'in_progress' ? 'warning' : 'pending'}
                  >
                    {weekLabel}
                  </Badge>
                </div>
                
                {/* 任务列表 */}
                <div className="space-y-3 border-l-2 border-border-default pl-6">
                  {weekTasks.map(task => (
                    <Card key={task.id} className="p-4" hover={false}>
                      <div className="flex items-start gap-3">
                        <span className="text-2xl">{getStatusIcon(task.status)}</span>
                        <div className="flex-1">
                          <h4 className="font-medium text-text-primary">{task.title}</h4>
                          {task.status === 'skipped' && (
                            <p className="mt-1 text-sm text-text-tertiary">已跳过</p>
                          )}
                        </div>
                      </div>
                    </Card>
                  ))}
                </div>
              </div>
            )
          })}
        </div>
      </div>

      {/* 学习统计 */}
      <div>
        <h2 className="mb-6 text-2xl font-semibold text-text-primary">学习统计</h2>
        <div className="grid grid-cols-1 gap-6 md:grid-cols-4">
          <Card className="p-6">
            <div className="text-sm text-text-secondary">学习天数</div>
            <div className="mt-2 text-4xl font-bold text-text-primary">{stats.daysSinceStart}</div>
            <Badge variant="success" className="mt-3">坚持中</Badge>
          </Card>
          
          <Card className="p-6">
            <div className="text-sm text-text-secondary">完成任务</div>
            <div className="mt-2 text-4xl font-bold text-text-primary">{completedTasks}/{tasks.length}</div>
            <Badge variant="info" className="mt-3">{Math.round(completedTasks / tasks.length * 100)}%</Badge>
          </Card>
          
          <Card className="p-6">
            <div className="text-sm text-text-secondary">知识卡片</div>
            <div className="mt-2 text-4xl font-bold text-text-primary">{stats.cardsCount}</div>
            <Badge variant="info" className="mt-3">持续增长</Badge>
          </Card>
          
          <Card className="p-6">
            <div className="text-sm text-text-secondary">活跃项目</div>
            <div className="mt-2 text-4xl font-bold text-text-primary">{stats.projectsCount}</div>
            <Badge variant="warning" className="mt-3">进行中</Badge>
          </Card>
        </div>
      </div>
    </div>
  )
}
```

```typescript
import { Badge } from '@/components/ui/Badge'
import { Card } from '@/components/ui/Card'
import { ProgressBar } from '@/components/ui/ProgressBar'
import vaultData from '@/../../data/vault-data.json'

// 状态映射
const statusMap = {
  'in_progress': { label: '进行中', variant: 'warning' as const, icon: '🟢' },
  'completed': { label: '已完成', variant: 'success' as const, icon: '✅' },
  'planned': { label: '计划中', variant: 'pending' as const, icon: '📋' },
  'paused': { label: '暂停', variant: 'pending' as const, icon: '⏸️' },
  'active': { label: '活跃', variant: 'success' as const, icon: '🟢' },
}

export default function ProjectsPage() {
  // 过滤掉模板项目
  const projects = vaultData.projects.filter(p => 
    !p.name.includes('{{') && 
    !p.name.includes('Example') &&
    p.name !== 'Example Project'
  )

  return (
    <div className="mx-auto max-w-7xl px-6 py-12">
      {/* 页面标题 */}
      <div className="mb-12">
        <h1 className="text-4xl font-bold text-text-primary">我的项目</h1>
        <p className="mt-4 text-lg text-text-secondary">
          从 Obsidian Vault 自动同步的项目列表
        </p>
      </div>

      {/* 项目网格 */}
      <div className="grid grid-cols-1 gap-6 md:grid-cols-2 lg:grid-cols-3">
        {projects.map(project => {
          const statusInfo = statusMap[project.status as keyof typeof statusMap] || statusMap.planned
          const hasProgress = project.totalTasks > 0

          return (
            <Card key={project.id} className="p-6">
              {/* 项目状态 */}
              <div className="mb-4 flex items-center justify-between">
                <Badge variant={statusInfo.variant}>
                  {statusInfo.icon} {statusInfo.label}
                </Badge>
                {project.tags && project.tags.length > 0 && (
                  <div className="flex gap-1">
                    {project.tags.slice(0, 2).map(tag => (
                      <span key={tag} className="text-xs text-text-tertiary">
                        #{tag}
                      </span>
                    ))}
                  </div>
                )}
              </div>

              {/* 项目名称 */}
              <h3 className="text-xl font-semibold text-text-primary">
                {project.name}
              </h3>

              {/* 项目描述 */}
              {project.description && (
                <p className="mt-3 line-clamp-3 text-sm leading-relaxed text-text-secondary">
                  {project.description}
                </p>
              )}

              {/* 进度条 */}
              {hasProgress && (
                <div className="mt-4">
                  <ProgressBar 
                    value={project.tasksCompleted} 
                    max={project.totalTasks}
                    size="sm"
                  />
                </div>
              )}

              {/* 项目日期 */}
              {project.startDate && (
                <div className="mt-4 text-xs text-text-tertiary">
                  开始于 {project.startDate}
                </div>
              )}
            </Card>
          )
        })}
      </div>

      {/* 空状态 */}
      {projects.length === 0 && (
        <div className="py-20 text-center">
          <p className="text-text-secondary">暂无项目</p>
        </div>
      )}
    </div>
  )
}
```

```typescript
'use client'

import { useState } from 'react'
import { Badge } from '@/components/ui/Badge'
import { Card } from '@/components/ui/Card'
import vaultData from '@/../../data/vault-data.json'

// 类型图标映射
const typeIcons: Record<string, string> = {
  'insight': '💡',
  'tip': '📌',
  'counterintuitive': '🔄',
  'concept': '🎯',
  'person': '👤',
  'book': '📖',
  'mentalmodel': '🧠',
  'card': '📝',
}

// 类型标签
const typeLabels: Record<string, string> = {
  'insight': '洞察',
  'tip': '技巧',
  'counterintuitive': '反直觉',
  'concept': '概念',
  'person': '人物',
  'book': '书籍',
  'mentalmodel': '心智模型',
  'card': '笔记',
}

export default function KnowledgePage() {
  const [selectedType, setSelectedType] = useState<string>('all')
  
  // 过滤示例卡片
  const allCards = vaultData.cards.filter(c => !c.title.includes('_EXAMPLE_'))
  
  // 按类型筛选
  const filteredCards = selectedType === 'all' 
    ? allCards 
    : allCards.filter(c => c.type === selectedType)
  
  // 获取所有类型
  const types = Array.from(new Set(allCards.map(c => c.type)))

  return (
    <div className="mx-auto max-w-7xl px-6 py-12">
      {/* 页面标题 */}
      <div className="mb-12">
        <h1 className="text-4xl font-bold text-text-primary">知识网络</h1>
        <p className="mt-4 text-lg text-text-secondary">
          共 {allCards.length} 张知识卡片
        </p>
      </div>

      {/* 类型筛选 */}
      <div className="mb-8 flex flex-wrap gap-3">
        <button
          onClick={() => setSelectedType('all')}
          className={`rounded-full px-4 py-2 text-sm font-medium transition-colors ${
            selectedType === 'all'
              ? 'bg-primary text-white'
              : 'bg-dark-elevated text-text-secondary hover:bg-dark-hover'
          }`}
        >
          全部 ({allCards.length})
        </button>
        {types.map(type => {
          const count = allCards.filter(c => c.type === type).length
          return (
            <button
              key={type}
              onClick={() => setSelectedType(type)}
              className={`rounded-full px-4 py-2 text-sm font-medium transition-colors ${
                selectedType === type
                  ? 'bg-primary text-white'
                  : 'bg-dark-elevated text-text-secondary hover:bg-dark-hover'
              }`}
            >
              {typeIcons[type] || '📝'} {typeLabels[type] || type} ({count})
            </button>
          )
        })}
      </div>

      {/* 卡片网格 */}
      <div className="grid grid-cols-1 gap-6 md:grid-cols-2 lg:grid-cols-3">
        {filteredCards.map(card => (
          <Card key={card.id} className="p-6">
            {/* 卡片类型 */}
            <div className="mb-3 flex items-center gap-2">
              <span className="text-2xl">{typeIcons[card.type] || '📝'}</span>
              <Badge variant="info" size="sm">
                {typeLabels[card.type] || card.type}
              </Badge>
            </div>

            {/* 卡片标题 */}
            <h3 className="text-lg font-semibold leading-snug text-text-primary">
              {card.title.replace(/^(insight|tip|counterintuitive|concept|person|book|mentalmodel|card)_/, '')}
            </h3>

            {/* 卡片摘要 */}
            {card.summary && (
              <p className="mt-3 line-clamp-3 text-sm leading-relaxed text-text-secondary">
                {card.summary}
              </p>
            )}

            {/* 标签 */}
            {card.tags && card.tags.length > 0 && (
              <div className="mt-4 flex flex-wrap gap-2">
                {card.tags.slice(0, 3).map(tag => (
                  <span 
                    key={tag}
                    className="rounded-full bg-dark-hover px-2 py-1 text-xs text-text-tertiary"
                  >
                    #{tag}
                  </span>
                ))}
              </div>
            )}

            {/* 创建日期 */}
            {card.createdAt && (
              <div className="mt-4 text-xs text-text-tertiary">
                {card.createdAt}
              </div>
            )}
          </Card>
        ))}
      </div>

      {/* 空状态 */}
      {filteredCards.length === 0 && (
        <div className="py-20 text-center">
          <p className="text-text-secondary">该类型暂无卡片</p>
        </div>
      )}
    </div>
  )
}
```
