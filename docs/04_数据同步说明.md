## 脚本实现详解

### vault-sync.js 核心代码

```javascript
#!/usr/bin/env node

const fs = require('fs')
const path = require('path')
const matter = require('gray-matter')

// 配置
const VAULT_PATH = process.env.VAULT_PATH || './vault'
const OUTPUT_PATH = process.env.OUTPUT_PATH || './data'

// 主函数
async function main() {
  console.log('🔄 开始同步 Vault...\n')
  
  try {
    // 1. 解析个人信息
    console.log('📝 解析个人信息...')
    const profile = parseProfile()
    
    // 2. 解析每日记录
    console.log('📅 解析每日记录...')
    const dailyLogs = parseDailyLogs(30) // 最近 30 天
    
    // 3. 解析项目
    console.log('📂 解析项目...')
    const projects = parseProjects()
    
    // 4. 解析知识卡片
    console.log('💡 解析知识卡片...')
    const cards = parseCards()
    
    // 5. 计算统计数据
    console.log('📊 计算统计数据...')
    const stats = calculateStats(dailyLogs, projects, cards)
    
    // 6. 生成知识图谱
    console.log('🕸️ 生成知识图谱...')
    const graph = generateKnowledgeGraph(cards)
    
    // 7. 输出 JSON
    console.log('💾 写入数据文件...\n')
    writeJSON('vault-data.json', {
      profile,
      dailyLogs,
      projects,
      cards,
      lastSync: new Date().toISOString()
    })
    
    writeJSON('learning-stats.json', stats)
    writeJSON('knowledge-graph.json', graph)
    
    // 8. 输出统计
    console.log('✅ 同步完成！\n')
    console.log('📊 统计：')
    console.log(`  - 每日记录: ${dailyLogs.length} 条`)
    console.log(`  - 项目: ${projects.length} 个`)
    console.log(`  - 知识卡片: ${cards.length} 张`)
    console.log(`  - 完成任务: ${stats.tasksCompleted}/${stats.totalTasks}`)
    
  } catch (error) {
    console.error('❌ 同步失败:', error.message)
    process.exit(1)
  }
}

// 解析个人信息
function parseProfile() {
  const filePath = path.join(VAULT_PATH, '01_Context/About_Me.md')
  
  if (!fs.existsSync(filePath)) {
    console.warn('⚠️  未找到 About_Me.md')
    return {
      name: '未知',
      location: '未知',
      bio: ''
    }
  }
  
  const fileContent = fs.readFileSync(filePath, 'utf8')
  const { data, content } = matter(fileContent)
  
  return {
    name: data.name || '未知',
    location: data.location || '未知',
    startDate: data.start_date || null,
    bio: data.bio || '',
    content: content
  }
}

// 解析每日记录
function parseDailyLogs(daysCount = 30) {
  const dailyDir = path.join(VAULT_PATH, '02_Daily')
  
  if (!fs.existsSync(dailyDir)) {
    console.warn('⚠️  未找到 02_Daily 目录')
    return []
  }
  
  const files = fs.readdirSync(dailyDir)
    .filter(f => f.match(/^\d{4}-\d{2}-\d{2}\.md$/))
    .sort()
    .reverse() // 最新的在前
    .slice(0, daysCount)
  
  return files.map(filename => {
    const filePath = path.join(dailyDir, filename)
    const fileContent = fs.readFileSync(filePath, 'utf8')
    const { data, content } = matter(fileContent)
    
    return {
      date: filename.replace('.md', ''),
      title: data.title || `${filename} 的记录`,
      tags: data.tags || [],
      highlights: data.highlights || [],
      content: content,
      tasksCompleted: countCompletedTasks(content),
      cardsCreated: data.cards_created || 0
    }
  })
}

// 解析项目
function parseProjects() {
  const projectsDir = path.join(VAULT_PATH, '03_Projects')
  
  if (!fs.existsSync(projectsDir)) {
    console.warn('⚠️  未找到 03_Projects 目录')
    return []
  }
  
  const projectDirs = fs.readdirSync(projectsDir, { withFileTypes: true })
    .filter(dirent => dirent.isDirectory())
  
  const projects = []
  
  for (const dir of projectDirs) {
    const overviewPath = path.join(projectsDir, dir.name, '00_Overview.md')
    
    if (!fs.existsSync(overviewPath)) continue
    
    const fileContent = fs.readFileSync(overviewPath, 'utf8')
    const { data, content } = matter(fileContent)
    
    // 提取任务列表
    const tasks = extractTasks(content)
    
    projects.push({
      id: dir.name,
      name: data.title || dir.name,
      status: data.status || 'planned',
      startDate: data.start_date || data.created || null,
      progress: data.progress || 0,
      tags: data.tags || [],
      description: data.description || '',
      content: content,
      tasks: tasks,
      tasksCompleted: tasks.filter(t => t.completed).length,
      totalTasks: tasks.length
    })
  }
  
  // 排序：进行中 → 已完成 → 计划中
  const statusOrder = { in_progress: 0, completed: 1, planned: 2 }
  projects.sort((a, b) => statusOrder[a.status] - statusOrder[b.status])
  
  return projects
}

// 解析知识卡片
function parseCards() {
  const cardsDir = path.join(VAULT_PATH, '04_Knowledge/00_Cards')
  
  if (!fs.existsSync(cardsDir)) {
    console.warn('⚠️  未找到 04_Knowledge/00_Cards 目录')
    return []
  }
  
  const files = fs.readdirSync(cardsDir)
    .filter(f => f.endsWith('.md') && !f.startsWith('.'))
  
  return files.map(filename => {
    const filePath = path.join(cardsDir, filename)
    const fileContent = fs.readFileSync(filePath, 'utf8')
    const { data, content } = matter(fileContent)
    
    return {
      id: filename.replace('.md', ''),
      type: data.type || 'note',
      title: data.title || filename.replace('.md', ''),
      tags: data.tags || [],
      createdAt: data.created || null,
      relatedCards: data.related_cards || [],
      content: content,
      summary: content.substring(0, 200).replace(/\n/g, ' ') + '...'
    }
  }).sort((a, b) => {
    // 按创建时间倒序
    if (!a.createdAt) return 1
    if (!b.createdAt) return -1
    return b.createdAt.localeCompare(a.createdAt)
  })
}

// 计算统计数据
function calculateStats(dailyLogs, projects, cards) {
  const learningProject = projects.find(p => p.id === '学习_Claude_Code')
  
  // 计算开始天数
  const startDate = new Date('2026-06-01') // 替换为实际开始日期
  const today = new Date()
  const daysSinceStart = Math.floor((today - startDate) / (1000 * 60 * 60 * 24))
  
  return {
    daysSinceStart: daysSinceStart,
    currentDay: daysSinceStart + 1,
    currentWeek: Math.ceil((daysSinceStart + 1) / 7),
    tasksCompleted: learningProject?.tasksCompleted || 0,
    totalTasks: learningProject?.totalTasks || 9,
    projectsCount: projects.length,
    cardsCount: cards.length,
    totalLearningHours: calculateTotalHours(dailyLogs),
    notesCount: dailyLogs.length
  }
}

// 生成知识图谱
function generateKnowledgeGraph(cards) {
  const nodes = cards.map(card => ({
    id: card.id,
    label: card.title,
    type: card.type,
    tags: card.tags
  }))
  
  const edges = []
  cards.forEach(card => {
    card.relatedCards.forEach(relatedId => {
      edges.push({
        source: card.id,
        target: relatedId,
        type: 'related'
      })
    })
  })
  
  return { nodes, edges }
}

// 工具函数：统计完成的任务
function countCompletedTasks(content) {
  const matches = content.match(/- \[x\]/gi)
  return matches ? matches.length : 0
}

// 工具函数：提取任务列表
function extractTasks(content) {
  const taskRegex = /- \[([ x])\] (.+)/gi
  const tasks = []
  let match
  
  while ((match = taskRegex.exec(content)) !== null) {
    tasks.push({
      title: match[2],
      completed: match[1].toLowerCase() === 'x'
    })
  }
  
  return tasks
}

// 工具函数：计算学习时长
function calculateTotalHours(dailyLogs) {
  // 简单估算：每天 1-2 小时
  return dailyLogs.length * 1.5
}

// 工具函数：写入 JSON
function writeJSON(filename, data) {
  const outputPath = path.join(OUTPUT_PATH, filename)
  fs.mkdirSync(OUTPUT_PATH, { recursive: true })
  fs.writeFileSync(outputPath, JSON.stringify(data, null, 2), 'utf8')
}

// 运行
main()
```

## 同步策略

### 增量同步 vs 全量同步

**当前采用**：全量同步（每次重新扫描所有文件）

**优点**：
- 实现简单
- 确保数据一致
- 不需要维护状态

**缺点**：
- 随着 Vault 增大，同步时间增长

**未来优化**：增量同步
- 记录文件 mtime（修改时间）
- 只处理变化的文件
- 维护索引文件

### 同步时机

1. **定时同步**：每天固定时间（例如凌晨 2:00）
2. **手动触发**：GitHub Actions workflow_dispatch
3. **Vault 更新触发**（可选）：Vault 仓库 push 时触发 webhook

### 冲突处理

由于 Vault 是单向数据源（Vault → 网站），不存在冲突问题。

## 性能优化

### 当前性能

- **文件数量**：约 50-100 个
- **同步时间**：约 1-2 秒
- **内存占用**：约 50 MB

### 优化建议

1. **缓存解析结果**：使用 mtime 判断是否需要重新解析
2. **并行处理**：使用 `Promise.all()` 并行解析多个文件
3. **流式处理**：对于大文件使用流式读取

```javascript
// 并行解析示例
async function parseProjects() {
  const projectDirs = getProjectDirs()
  
  const projects = await Promise.all(
    projectDirs.map(dir => parseProject(dir))
  )
  
  return projects.filter(p => p !== null)
}
```
